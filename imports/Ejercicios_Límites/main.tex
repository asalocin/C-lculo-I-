\chapter{Axiomas de Orden}

\begin{prob}{}{}
    \begin{itemize}
        \item[\textbf{a)}] $\displaystyle \lim_{x\to 0} \left(1 + \frac{x}{x^2+1}\right)^{\frac{1}{x}}$ 
        \item[\textbf{b)}] $\displaystyle \lim_{x \to -5} \frac{x^2-25}{x^2+2x-15}$ 
        \item[\textbf{c)}] $\displaystyle \lim_{x \to 0} \frac{\sin(\alpha x )}{\sin(\beta x)}$ \quad Si $\alpha,\beta \in \mathbb{R}$.
        \item[\textbf{d)}] $\displaystyle \lim_{x \to 0} x^2 \sin\left(\frac{1}{x}\right)$

        \item[\textbf{f)}] $\displaystyle \lim_{x \to a } \frac{\sqrt[n]{x}-\sqrt[n]{a}}{\sqrt{x}-\sqrt{a}}$
    \end{itemize}
\end{prob}
\textbf{Solución:}
\begin{itemize}
    \item[\textbf{a)}] Recordemos que $x = e^{\ln(x)}$. Entonces
    \begin{align*}
        \lim_{x\to 0} \left(1 + \frac{x}{x^2+1}\right)^{\frac{1}{x}} &=  \lim_{x\to 0}  \exp \left(\ln\left(1 + \frac{x}{x^2+1}\right)^{\frac{1}{x}} \right) \\
        &=\lim_{x\to 0} \exp \left(\frac{\ln\left(1 + \frac{x}{x^2+1}\right)}{x}\right) \\ 
        &=\lim_{x\to 0} \exp \left(\frac{\ln\left(1 + \frac{x}{x^2+1}\right)}{x} \cdot \frac{\frac{x}{x^2+1}}{\frac{x}{x^2+1}}\right) \\ 
        &= \lim_{x\to 0}  \exp \left(\frac{\ln\left(1 + \frac{x}{x^2+1}\right)}{\frac{x}{x^2+1}} \cdot \frac{1}{x^2+1}\right) \\
        &= \exp \left(\lim_{x\to 0}\left(\frac{\ln\left(1 + \frac{x}{x^2+1}\right)}{\frac{x}{x^2+1}} \cdot \frac{1}{x^2+1}\right) \right) && \textbf{(Continuidad)} \\
        &= \exp( 1\cdot 1) \\
        &=e
    \end{align*}

    \item[\textbf{b)}] Factoricemos el numerador y denominador:

    \begin{align*}
        \lim_{x\to -5 }\frac{x^2-25}{x^2+2x-15} &= \lim_{x \to -5}\frac{(x-5)(x+5)}{(x-3)(x+5)} \\
        &=\lim_{x \to -5}\frac{x-5}{x-3} \\
        &= \frac{-5-5}{-5-3} \\
        &=\frac{5}{4}
    \end{align*}

    \item[\textbf{c)}] Recordemos el límite $\displaystyle \lim_{x \to 0} \frac{\sin(x)}{x}=1$. Luego
    \begin{align*}
       \lim_{x \to 0} \frac{\sin(\alpha x )}{\sin(\beta x)}&=\lim_{x \to 0} \frac{\sin(\alpha x)}{\sin(\beta x )}\cdot \frac{\alpha \beta }{\alpha \beta} \\
       &= \lim_{x \to 0} \underbrace{\frac{\sin(\alpha x ) }{\alpha}}_{1} \cdot \underbrace{\frac{\beta }{\sin(\beta x )}}_{1} \cdot \frac{\alpha}{\beta} \\
       &= \frac{\alpha}{\beta}
    \end{align*}

    \item[\textbf{d)}] Recordemos que $|\sin(x)| \leq 1 $ $\forall x \in \mathbb{R}$. Entonces
    \begin{align*}
       \left| x^2 \sin\left(\frac{1}{x}\right) \right|\leq |x^2|
    \end{align*}
    Es decir
    \begin{align*}
        -x^2\leq x^2 \sin\left(\frac{1}{x}\right) \leq x^2 
    \end{align*}
    Dado que 
    \begin{align*}
        \lim_{x\to 0}x^2=\lim_{x\to 0}-x^2=0
    \end{align*}
Por \textbf{Teorema del Sandwich} concluimos que 
\begin{align*}
    \lim_{x \to 0} x^2 \sin\left(\frac{1}{x}\right)=0
\end{align*}
